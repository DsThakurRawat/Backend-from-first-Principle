//technique 2:field and method promotion
// when you embed a struct , all of its fields and method are promoted to the outer struct.You cna access them directly as if they belong to the outer struct
emp := Employee{}
// We can access Name directly because it was PROMOTED from Person
emp.Name = "Divyans"

// Or we can access it explicitly through the embedded type
emp.Person.Name = "Divyansh"
